export const TabsStory = (args) => (

          <TabsTrigger
            {...args}
            value={`tab${item}`}

            rightIcon={
              <Badge
                count="99"

              />

          </TabsTrigger>
        );
      })}
    </TabsList>
    {Array.from(Array(100).keys()).map((item) => {
      return (
        <TabsContent className="TabsPrimitiveContent" value={`tab${item}`}>
          <p>content tab {item}</p>
        </TabsContent>
      );
    })}
  </Tabs>
);
