## Example Usage

```tsx
import { InputNumber } from '@zero-ui/core';
```

<Canvas columns={2} withSource="close">
  <Story
    name="InputNumber"
    args={{
      defaultValue: '',
      placeHolder: 'input number',
      leftIcon: <PlusOutlined />,
      decrementIcon: <DownOutlined />,
      incrementIcon: <UpOutlined />,
      rightIcon: '$',
      max: 5,
      min: 0,
      enableNegativeValue: false,
    }}
  >
    {(args) => {
      const [value, setValue] = useState(0);
      return (
        <InputNumber
          value={value}
          onChange={(e) => setValue(e.target.value)}
          decrement={() => setValue(value + 1)}
          increment={() => setValue(value - 1)}
          {...args}
        />
      );
    }}
  </Story>
</Canvas>
